function PhotoPicker({photos,setPhotos,max=MAX_PHOTOS}){
  const [busy,setBusy]=useState(false); const fileRef=useRef();

  async function onFiles(e){
    const files=Array.from(e.target.files||[]).slice(0,remaining);
    if(!files.length)return; setBusy(true);
    for(const f of files){ try{ const url=await compressImage(f); setPhotos((prev)=> prev.length>=max?prev:[...prev,{id:uid(),url,isNew:true}]); }catch(err){} }
    setBusy(false); if(fileRef.current) fileRef.current.value="";
  }
  return (
    <div style={{marginBottom:18}}>

function speakNow(text, mutedRef){
  if(mutedRef.current) return null;
  try{
    if(!("speechSynthesis" in window)) return null;
    const u=new SpeechSynthesisUtterance(text);
    u.lang="ja-JP"; u.rate=1.05; u.pitch=1;
    window.speechSynthesis.speak(u); return u;
  }catch(e){ return null; }
}
function Timer({plan,onToggle,palette}){
  const blocks=plan.blocks||[];
  const declareText=plan.declareText!==undefined?plan.declareText:DECLARE_TEXT;
  const [phase,setPhase]=useState("idle"); // idle|countin|declaring|running|costume-intro|costume|done
  const [countin,setCountin]=useState(60);
  const [elapsed,setElapsed]=useState(0);
  const [cElapsed,setCElapsed]=useState(0);
  const [paused,setPaused]=useState(false);
  const [muted,setMuted]=useState(false);
  const [planRead,setPlanRead]=useState(true);

  const phaseRef=useRef("idle"), countinRef=useRef(60), elapsedRef=useRef(0), cElapsedRef=useRef(0);
  const pausedRef=useRef(false), mutedRef=useRef(false), firedRef=useRef(new Set()), tickRef=useRef(null);
  const planReadRef=useRef(true), blocksRef=useRef(blocks);
  const mainSecRef=useRef(0), costumeSecRef=useRef(0), stepsRef=useRef([]);
  useEffect(()=>{ blocksRef.current=blocks; },[blocks]);
  const setP=(p)=>{ phaseRef.current=p; setPhase(p); };
  const clearTick=()=>{ if(tickRef.current){ clearInterval(tickRef.current); tickRef.current=null; } };
  const say=(text)=>speakNow(text,mutedRef);
  useEffect(()=>()=>{ clearTick(); try{ if("speechSynthesis" in window) window.speechSynthesis.cancel(); }catch(e){} },[]);

  const mainSecCalc=flatMainSteps(blocks).reduce((a,s)=>a+s.min*60,0);
  const costumeSecCalc=blocks.filter((b)=>b.type==="costume"&&b.on).reduce((a,b)=>a+blockMin(b)*60,0);
  const canStart=mainSecCalc>0||costumeSecCalc>0;

  function lockDurations(){
    const bl=blocksRef.current||[];
    stepsRef.current=flatMainSteps(bl);
    mainSecRef.current=stepsRef.current.reduce((a,s)=>a+s.min*60,0);
    costumeSecRef.current=bl.filter((b)=>b.type==="costume"&&b.on).reduce((a,b)=>a+blockMin(b)*60,0);
  }
  function begin(){ lockDurations(); if(mainSecRef.current>0) startCountin(); else if(costumeSecRef.current>0) startCostume(); }

  function speakDeclare(){ if(mutedRef.current||!("speechSynthesis" in window)||!declareText.trim())return; try{ const u=new SpeechSynthesisUtterance(declareText); u.lang="ja-JP"; u.rate=1.05; window.speechSynthesis.speak(u); }catch(e){} }
  function startCountin(){
    clearTick(); try{ if("speechSynthesis" in window){ window.speechSynthesis.cancel(); window.speechSynthesis.speak(new SpeechSynthesisUtterance(" ")); } }catch(e){}
    firedRef.current=new Set(); countinRef.current=60; setCountin(60); setP("countin"); pausedRef.current=false; setPaused(false);
    say("1分前");
    tickRef.current=setInterval(()=>{
      countinRef.current-=1; const c=countinRef.current;
      if(c===30) say("30秒前"); else if(c===10) say("10秒前");
      if(c===DECLARE_LEAD) speakDeclare();          // 残り5秒で部門名
      if(c<=0){ clearTick(); setCountin(0); say("競技開始"); startRunning(); return; } // 0で「競技開始」＆スタート
      setCountin(c);
    },1000);
  }
  function declareThenStart(){
    setP("declaring");
    let started=false; const go=()=>{ if(started)return; started=true; say("競技開始"); startRunning(); };
    if(mutedRef.current||!("speechSynthesis" in window)){ setTimeout(go,700); return; }
    try{ const u=new SpeechSynthesisUtterance(declareText); u.lang="ja-JP"; u.rate=1.05; u.onend=go; u.onerror=go; window.speechSynthesis.speak(u); setTimeout(()=>{ if(phaseRef.current==="declaring") go(); },9000); }catch(e){ go(); }
  }
  function startRunning(){
    clearTick(); firedRef.current=new Set(); elapsedRef.current=0; setElapsed(0); setP("running"); pausedRef.current=false; setPaused(false);
    const MS=mainSecRef.current;
    tickRef.current=setInterval(()=>{
      if(pausedRef.current) return;
      elapsedRef.current+=1; const e=elapsedRef.current, remain=MS-e;
      const F=firedRef.current; const one=(k,t)=>{ if(!F.has(k)){ F.add(k); say(t); } };
      if(planReadRef.current){ const st=stepsRef.current||[]; let acc=0; for(let i=0;i<st.length;i++){ if(i>0 && e===acc){ const cur=st[i], prev=st[i-1]; one("st"+i,(cur.block!==prev.block?cur.block+"、":"")+(cur.name||"次の工程")+"に入ります。"); } acc+=st[i].min*60; } }
      if(e>0 && e%600===0 && e<MS) one("e"+e, (e/60)+"分経過。残り"+Math.round((MS-e)/60)+"分です。");
      if(remain===180) one("r180","3分前。");
      if(remain===60) one("r60","1分前。");
      if(remain===30) one("r30","30秒前。");
      if(remain===10) one("r10","10秒前。");
      if(remain<=5 && remain>=1) one("r"+remain, String(remain));
      if(e>=MS){ clearTick(); setElapsed(MS); one("end","時間です。"); afterMain(); return; }
      setElapsed(e);
    },1000);
  }
  function afterMain(){ if(costumeSecRef.current>0){ setTimeout(()=>startCostume(),1400); } else setP("done"); }
  function startCostume(){
    clearTick(); firedRef.current=new Set(); cElapsedRef.current=0; setCElapsed(0); setP("costume-intro"); pausedRef.current=false; setPaused(false);
    const beginCount=()=>{ if(phaseRef.current==="done") return; startCostumeCount(); };
    const sayHajime=()=>{
      if(mutedRef.current||!("speechSynthesis" in window)){ setTimeout(beginCount,400); return; }
      try{ const u=new SpeechSynthesisUtterance("はじめ"); u.lang="ja-JP"; u.rate=1.05; let d=false; const g=()=>{ if(d)return; d=true; beginCount(); }; u.onend=g; u.onerror=g; window.speechSynthesis.speak(u); setTimeout(g,2500); }catch(e){ beginCount(); }
    };
    if(mutedRef.current||!("speechSynthesis" in window)){ setTimeout(sayHajime,600); return; }
    try{ const u=new SpeechSynthesisUtterance("1分間の衣装装着に入ります。"); u.lang="ja-JP"; u.rate=1.05; let d=false; const g=()=>{ if(d)return; d=true; sayHajime(); }; u.onend=g; u.onerror=g; window.speechSynthesis.speak(u); setTimeout(g,5000); }catch(e){ sayHajime(); }
  }
  function startCostumeCount(){
    clearTick(); setP("costume"); const CS=costumeSecRef.current; cElapsedRef.current=0; setCElapsed(0); firedRef.current=new Set();
    tickRef.current=setInterval(()=>{
      if(pausedRef.current) return;
      cElapsedRef.current+=1; const ce=cElapsedRef.current, cr=CS-ce;
      const F=firedRef.current; const one=(k,t)=>{ if(!F.has(k)){ F.add(k); say(t); } };
      if(cr===30) one("c30","残り30秒。");
      if(cr===10) one("c10","残り10秒。");
      if(cr<=5 && cr>=1) one("c"+cr, String(cr));
      if(ce>=CS){ clearTick(); setCElapsed(CS); one("cend","やめ。"); setP("done"); return; }
      setCElapsed(ce);
    },1000);
  }
  function togglePause(){ pausedRef.current=!pausedRef.current; setPaused(pausedRef.current); }
  function reset(){ clearTick(); try{ if("speechSynthesis" in window) window.speechSynthesis.cancel(); }catch(e){} firedRef.current=new Set(); elapsedRef.current=0; cElapsedRef.current=0; countinRef.current=60; setElapsed(0); setCElapsed(0); setCountin(60); pausedRef.current=false; setPaused(false); setP("idle"); }
  function skip(){ clearTick(); declareThenStart(); }
  function toggleMute(){ const n=!mutedRef.current; mutedRef.current=n; setMuted(n); if(n){ try{ if("speechSynthesis" in window) window.speechSynthesis.cancel(); }catch(e){} } }
  function togglePlanRead(){ const n=!planReadRef.current; planReadRef.current=n; setPlanRead(n); }

  const MS=mainSecRef.current||mainSecCalc;
  const remain=MS-elapsed;
  const cRemain=(costumeSecRef.current||costumeSecCalc)-cElapsed;
  let curName=null, curBlock=null;
  if(phase==="running"){ const st=stepsRef.current||[]; let acc=0; for(let i=0;i<st.length;i++){ const m=st[i].min*60; if(elapsed<acc+m){ curName=st[i].name; curBlock=st[i].block; break; } acc+=m; } }

  let big="0:00", subline=null, bigColor=C.ink, pulse=false;
  if(phase==="idle"){ big=mmss(mainSecCalc||costumeSecCalc); subline=canStart?(costumeSecCalc>0&&mainSecCalc>0?`本番 ${mainSecCalc/60}分 ＋ 衣装 ${costumeSecCalc/60}分`:mainSecCalc>0?`本番 ${mainSecCalc/60}分`:`衣装装着 ${costumeSecCalc/60}分`):"大区分をオンにしてください"; }
  else if(phase==="countin"){ big=mmss(countin); subline=countin<=DECLARE_LEAD?(declareText||"まもなく競技開始"):"まもなく競技開始"; bigColor=C.primary; }
  else if(phase==="declaring"){ big="START"; subline=declareText||"競技開始"; bigColor=C.primary; }
  else if(phase==="running"){ big=mmss(remain); bigColor=remain<=60?C.alert:C.ink; pulse=remain<=60; subline=curName?<>いま：<span style={{color:C.primary,fontWeight:600}}>{curBlock}</span> / <span style={{fontWeight:600}}>{curName}</span></>:"計測中"; }
  else if(phase==="costume-intro"){ big="衣装"; subline="1分間の衣装装着に入ります"; bigColor=C.primary; }
  else if(phase==="costume"){ big=mmss(cRemain); bigColor=cRemain<=10?C.alert:C.ink; pulse=cRemain<=10; subline="衣装装着タイム"; }
  else if(phase==="done"){ big="0:00"; subline="お疲れさまでした"; bigColor=C.good; }

  const pct = phase==="running" ? elapsed/(MS||1)*100 : phase==="costume" ? cElapsed/((costumeSecRef.current||costumeSecCalc)||1)*100 : phase==="done" ? 100 : 0;
  const isBig=big==="START"||big==="衣装";

  return (
    <section style={{background:C.surface,border:`1px solid ${C.line}`,borderRadius:18,padding:"20px 18px 22px",boxShadow:C.shadow}}>
      <div style={{display:"flex",justifyContent:"space-between",alignItems:"center",gap:8,flexWrap:"wrap"}}>
        <div style={{fontSize:11.5,color:C.faint,letterSpacing:2,textTransform:"uppercase",fontFamily:"'Space Mono',monospace"}}>本番タイマー</div>
        <div style={{display:"flex",gap:8}}>
          <button onClick={togglePlanRead} aria-label="工程ガイドの読み上げ切替" style={{border:`1px solid ${C.line}`,background:planRead?C.primarySoft:C.surface2,color:planRead?C.primary:C.faint,borderRadius:9,padding:"5px 10px",fontSize:12.5,cursor:"pointer",fontWeight:600}}>◱ 工程 {planRead?"ON":"OFF"}</button>
          <button onClick={toggleMute} aria-label={muted?"読み上げオン":"読み上げオフ"} style={{border:`1px solid ${C.line}`,background:muted?C.surface2:C.primarySoft,color:muted?C.faint:C.primary,borderRadius:9,padding:"5px 10px",fontSize:12.5,cursor:"pointer",fontWeight:600}}>{muted?"🔇 消音":"🔊 音声"}</button>
        </div>
      </div>
      <div className={pulse?"pulse":""} style={{textAlign:"center",fontFamily:"'Fraunces',serif",fontSize:isBig?46:64,fontWeight:500,lineHeight:1.05,margin:"8px 0 4px",color:bigColor,fontVariantNumeric:"tabular-nums",letterSpacing:isBig?4:0}}>{big}</div>
      <div style={{textAlign:"center",fontSize:13.5,color:C.muted,minHeight:20,marginBottom:16,padding:"0 6px"}}>{subline}</div>
      <div style={{height:6,borderRadius:3,background:C.bg2,overflow:"hidden",marginBottom:18}}><div style={{height:"100%",width:`${pct}%`,background:phase==="done"?C.good:C.primary,transition:"width .9s linear"}}/></div>

      <div style={{display:"flex",gap:10}}>
        {phase==="idle" && <button onClick={begin} disabled={!canStart} style={{...primaryBtn,opacity:canStart?1:0.4,cursor:canStart?"pointer":"default"}}>{mainSecCalc>0?"本番開始":"衣装装着を開始"}</button>}
        {phase==="countin" && <><button onClick={skip} style={{...ghostBtn,flex:1,padding:14}}>スキップ</button><button onClick={reset} style={{...ghostBtn,flex:1,padding:14}}>中止</button></>}
        {phase==="declaring" && <button onClick={reset} style={{...ghostBtn,flex:1,padding:14}}>中止</button>}
        {phase==="running" && <><button onClick={togglePause} style={{flex:2,padding:14,border:"none",borderRadius:12,background:paused?C.primary:C.surface2,color:paused?"#fff":C.ink,fontSize:15,fontWeight:600,cursor:"pointer",fontFamily:"'Inter',sans-serif"}}>{paused?"再開":"一時停止"}</button><button onClick={reset} style={{...ghostBtn,flex:1,padding:14}}>リセット</button></>}
        {phase==="costume-intro" && <button onClick={reset} style={{...ghostBtn,flex:1,padding:14}}>中止</button>}
        {phase==="costume" && <><button onClick={togglePause} style={{flex:2,padding:14,border:"none",borderRadius:12,background:paused?C.primary:C.surface2,color:paused?"#fff":C.ink,fontSize:15,fontWeight:600,cursor:"pointer",fontFamily:"'Inter',sans-serif"}}>{paused?"再開":"一時停止"}</button><button onClick={reset} style={{...ghostBtn,flex:1,padding:14}}>リセット</button></>}
        {phase==="done" && <button onClick={reset} style={primaryBtn}>もう一度</button>}
      </div>

      {phase==="idle" && (
        <>
          <div style={{display:"flex",gap:8,marginTop:12}}>
            {blocks.map((b)=>(
              <button key={b.id} onClick={()=>onToggle(b.id)} aria-pressed={b.on} style={{flex:1,padding:"10px 4px",borderRadius:11,border:`1px solid ${b.on?C.primary:C.line}`,background:b.on?C.primarySoft:C.surface2,color:b.on?C.primary:C.faint,cursor:"pointer",fontFamily:"'Inter',sans-serif"}}>
                <div style={{fontSize:13,fontWeight:600}}>{b.name}</div>
                <div style={{fontSize:10.5,fontFamily:"'Space Mono',monospace",marginTop:3,opacity:0.85}}>{blockMin(b)}分 · {b.on?"ON":"OFF"}</div>
              </button>
            ))}
          </div>
          <p style={{fontSize:11,color:C.faint,textAlign:"center",margin:"12px 0 0",lineHeight:1.6}}>音量をオンにしてください。1分前→30秒前→10秒前→5秒前「第一部門クラシカルクリエイティブスタイル」→「競技開始」→（本番中は小工程・経過アナウンス）→3分前→…→5.4.3.2.1「時間です」→衣装は「はじめ」→残り30秒→残り10秒→5.4.3.2.1「やめ」</p>
        </>
      )}
    </section>
  );
}
